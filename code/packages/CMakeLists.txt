set(CPACK_PACKAGE_NAME "SPTK")
set(CPACK_PACKAGE_VENDOR "linotex.net")
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "Cross-platform development library")
set(CPACK_PACKAGE_VERSION ${VERSION})
set(CPACK_PACKAGE_VERSION_MAJOR ${VERSION_MAJOR})
set(CPACK_PACKAGE_VERSION_MINOR ${VERSION_MINOR})
set(CPACK_PACKAGE_VERSION_PATCH ${VERSION_PATCH})
set(CPACK_PACKAGE_INSTALL_DIRECTORY "packages")

#SET (CMAKE_INSTALL_RPATH /usr/local/lib)
#set(CPACK_DEBIAN_FILE_NAME DEB-DEFAULT)

IF (OS_FLAVOUR STREQUAL "Debian")
    SET (CPACK_GENERATOR "DEB")
    SET (CPACK_DEB_PACKAGE_COMPONENT ON)
    SET (CPACK_DEB_COMPONENT_INSTALL ON)
    SET (CPACK_DEBIAN_PACKAGE_SHLIBDEPS TRUE)
    SET (CPACK_DEBIAN_PACKAGE_GENERATE_SHLIBS ON)
    SET (CPACK_DEBIAN_PACKAGE_MAINTAINER "alexey.parshin@linotex.net")
ENDIF ()

set(CPACK_ARCHIVE_COMPONENT_INSTALL 1)

SET (CPACK_COMPONENTS_ALL sputil5 spdb5)
SET (CPACK_COMPONENT_SPDB5_DEPENDS sputil5)

IF (FLTK_LIBRARY)
    LIST (APPEND CPACK_COMPONENTS_ALL sptk5)
    SET (CPACK_COMPONENT_SPTK5_DEPENDS sputil5 spdb5)
ENDIF ()

IF (Oracle_LIBRARIES)
    LIST (APPEND CPACK_COMPONENTS_ALL spdb5_oracle)
    SET (CPACK_COMPONENT_SPDB5_ORACLE_DEPENDS sputil5 spdb5)
ENDIF ()

IF (PostgreSQL_LIBRARY)
    LIST (APPEND CPACK_COMPONENTS_ALL spdb5_postgresql)
    SET (CPACK_COMPONENT_SPDB5_POSTGRESQL_DEPENDS sputil5 spdb5)
ENDIF ()

IF (MySQL_LIBRARY)
    LIST (APPEND CPACK_COMPONENTS_ALL spdb5_mysql)
    SET (CPACK_COMPONENT_SPDB5_MYSQL_DEPENDS sputil5 spdb5)
ENDIF ()

IF (Firebird_LIBRARY)
    LIST (APPEND CPACK_COMPONENTS_ALL spdb5_firebird)
    SET (CPACK_COMPONENT_SPDB5_FIREBIRD_DEPENDS sputil5 spdb5)
ENDIF ()

IF (SQLITE3_LIBRARY)
    LIST (APPEND CPACK_COMPONENTS_ALL spdb5_sqlite3)
    SET (CPACK_COMPONENT_SPDB5_SQLITE3_DEPENDS sputil5 spdb5)
ENDIF ()

IF (ODBC_LIBRARY)
    LIST (APPEND CPACK_COMPONENTS_ALL spdb5_odbc)
    SET (CPACK_COMPONENT_SPDB5_ODBC_DEPENDS sputil5 spdb5)
ENDIF ()

set(CPACK_COMPONENT_SPUTIL5_DESCRIPTION "Core of SPTK library")
set(CPACK_COMPONENT_SPDB5_DESCRIPTION "SPTK database library")
set(CPACK_COMPONENT_SPDB5_MYSQL_DESCRIPTION "SPTK database library")
set(CPACK_COMPONENT_SPDB5_ODBC_DESCRIPTION "SPTK ODBC support")
set(CPACK_COMPONENT_SPDB5_ORACLE_DESCRIPTION "SPTK Oracle support")
set(CPACK_COMPONENT_SPDB5_POSTGRESQL_DESCRIPTION "SPTK PostgreSQL support")
set(CPACK_COMPONENT_SPDB5_SQLITE3_DESCRIPTION "SPTK SQLite3 support")
set(CPACK_COMPONENT_SPDB5_FIREBIRD_DESCRIPTION "SPTK Firebird support")
set(CPACK_COMPONENT_SPTK5_DESCRIPTION "SPTK GUI library")
